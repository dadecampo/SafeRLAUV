86 public static List<Mesh> GetSkinnedMeshBoneMeshes(GameObject obj)
87 {
88 List<Mesh> boneMeshes = new List<Mesh>();
89 List<List<int>> boneTriangles = new List<List<int>>();
90
91 // Get a reference to the mesh
92 var skinnedMeshRenderer = obj.GetComponent<SkinnedMeshRenderer>();
93 int boneCount = skinnedMeshRenderer.bones.Length;
94
95 if (boneCount == 0)
96 return boneMeshes;
97
98 var mesh = skinnedMeshRenderer.sharedMesh;
99
100 for (int i = 0; i < boneCount; i++)
101 {
102 Mesh bmesh = new Mesh();
103 bmesh.SetVertices(mesh.vertices);
104 boneMeshes.Add(bmesh);
105 boneTriangles.Add(new List<int>());
106 }
107
108 var bonesPerVertex = mesh.GetBonesPerVertex();
109 var boneWeights = mesh.GetAllBoneWeights();
110 var boneWeightIndex = 0;
111
112 List<int> VertexBones = new List<int>();
113
114 // Iterate over the vertices
115 for (var vertIndex = 0; vertIndex < mesh.vertexCount; vertIndex++)
116 {
117 var numberOfBonesForThisVertex = bonesPerVertex[vertIndex];
118
119 var currentBoneWeight = boneWeights[boneWeightIndex];
120 VertexBones.Add(currentBoneWeight.boneIndex);
121 boneWeightIndex += numberOfBonesForThisVertex;
122 }
123
124 var triangles = mesh.triangles;
125
126 // Iterate over triangles and add them to respective meshes depending on the bones
127 for (var triangle = 0; triangle < triangles.Length; triangle += 3)
128 {
129 int bone0 = VertexBones[triangles[triangle + 0]];
130 int bone1 = VertexBones[triangles[triangle + 1]];
131 int bone2 = VertexBones[triangles[triangle + 2]];
132
133 for (int i = 0; i < 3; i++)
134 {
135 boneTriangles[bone0].Add(triangles[triangle + i]);
136 }
137
138 if (bone1 != bone0)
139 for (int i = 0; i < 3; i++)
140 {
141 boneTriangles[bone1].Add(triangles[triangle + i]);
142 }
143
144 if (bone2 != bone1 && bone2 != bone0)
145 for (int i = 0; i < 3; i++)
146 {
147 boneTriangles[bone2].Add(triangles[triangle + i]);
148 }
149 }
150
151 for (int i = 0; i < boneCount; i++)
152 {
153 boneMeshes[i].SetTriangles(boneTriangles[i], 0);
154 boneMeshes[i] = ClearBlanks(boneMeshes[i]);
155 }
156
157 return boneMeshes;
158 }

162 // remove vertices which are not used by the triangles
163 private static Mesh ClearBlanks(Mesh mesh)
164 {
165 int[] triangles = mesh.triangles;
166 Vector3[] vertices = mesh.vertices;
167
168 List<Vector3> newVertList = new List<Vector3>();
169
170 List<int> oldVertNewID = new List<int>();
171 oldVertNewID.AddRange(Enumerable.Repeat(-1, vertices.Length));
172
173 List<int> trianglesList = triangles.ToList();
174
175 for (int i = 0; i < triangles.Length; i++)
176 {
177 int vertID = triangles[i];
178
179 if (oldVertNewID[vertID] == -1) // add vertex
180 {
181 oldVertNewID[vertID] = newVertList.Count;
182 newVertList.Add(vertices[vertID]);
183 }
184
185 trianglesList[i] = oldVertNewID[vertID];
186 }
187
188 triangles = trianglesList.ToArray();
189 vertices = newVertList.ToArray();
190 mesh.Clear();
191 mesh.vertices = vertices;
192 mesh.triangles = triangles;
193 return mesh;
194 }
