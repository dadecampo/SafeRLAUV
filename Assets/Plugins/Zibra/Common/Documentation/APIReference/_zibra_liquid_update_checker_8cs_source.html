206 private bool CheckIsLatestVersion(string pluginVersion, string localVersion)
207 {
208 char[] separator = { '.' };
209 int[] currentVersion = {};
210 int[] serverVersion = {};
211
212 try
213 {
214 string[] currentVersionStrArr = localVersion.Split(separator);
215 currentVersion = new int[currentVersionStrArr.Length];
216 for (int i = 0; i < currentVersionStrArr.Length; i++)
217 {
218 currentVersion[i] = int.Parse(currentVersionStrArr[i]);
219 }
220
221 string[] serverVersionStrArr = pluginVersion.Split(separator);
222 serverVersion = new int[serverVersionStrArr.Length];
223 for (int i = 0; i < serverVersionStrArr.Length; i++)
224 {
225 serverVersion[i] = int.Parse(serverVersionStrArr[i]);
226 }
227 }
228 catch (Exception)
229 {
230 return true;
231 }
232
233 if (currentVersion.Length < serverVersion.Length)
234 {
235 int[] tempArr = currentVersion;
236 currentVersion = new int[serverVersion.Length];
237 Array.Copy(tempArr, currentVersion, tempArr.Length);
238 }
239 else if (currentVersion.Length > serverVersion.Length)
240 {
241 int[] tempArr = serverVersion;
242 serverVersion = new int[currentVersion.Length];
243 Array.Copy(tempArr, serverVersion, tempArr.Length);
244 }
245
246 for (int i = 0; i < Math.Min(currentVersion.Length, serverVersion.Length); i++)
247 {
248 if (currentVersion[i] < serverVersion[i])
249 {
250 return false;
251 }
252 else if (currentVersion[i] > serverVersion[i])
253 {
254 return true;
255 }
256 }
257 return true;
258 }
