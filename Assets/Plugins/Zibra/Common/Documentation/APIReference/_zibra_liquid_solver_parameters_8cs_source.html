425 internal void ValidateParameters()
426 {
427 if (Application.isPlaying)
428 {
429 if (MaximumSpeciesCount == 0)
430 {
431 MaximumSpeciesCount = AdditionalParticleSpecies.Count + MAX_RUNTIME_ADDED_SPECIES;
432 }
433 else
434 {
435 if (AdditionalParticleSpecies.Count >= MaximumSpeciesCount)
436 {
437 int toDelete = AdditionalParticleSpecies.Count - MaximumSpeciesCount + 1;
438 AdditionalParticleSpecies.RemoveRange(MaximumSpeciesCount - 1, toDelete);
439 Debug.LogWarning("Zibra Liquids: Exceeded maximum number of species in current instance");
440 }
441 }
442 }
443
444 foreach (var param in AdditionalParticleSpecies)
445 {
446 int ZeroParameterCount = 0;
447 if (param.Gravity == Vector3.zero)
448 ZeroParameterCount++;
449 ValidateGravity(ref param.Gravity);
450 if (param.FluidStiffness == 0.0f)
451 {
452 ZeroParameterCount++;
453 param.FluidStiffness = DEFAULT_STIFFNESS;
454 }
455 if (param.ParticleDensity == 0.0f)
456 {
457 ZeroParameterCount++;
458 param.ParticleDensity = DEFAULT_DENSITY;
459 }
460 if (param.Viscosity == 0.0f)
461 ZeroParameterCount++;
462 if (param.MaximumVelocity == 0.0f)
463 ZeroParameterCount++;
464
465 if (ZeroParameterCount == 5) // if parameters are initialized as 0
466 {
467 param.MaximumVelocity = DEFAULT_MAX_VELOCITY;
468 param.Viscosity = DEFAULT_VISCOSITY;
469 param.Gravity = new Vector3(0.0f, DEFAULT_GRAVITY, 0.0f);
470 }
471 }
472
473 ValidateGravity(ref Gravity);
474 }
