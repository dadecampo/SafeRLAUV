731 private void StepPhysics()
732 {
733 solverCommandBuffer.Clear();
734
735 ForceCloseCommandEncoder(solverCommandBuffer);
736
737 SetSimulationParameters();
738
739 ManipulatorManager.UpdateDynamic(this, LastTimestep);
740
741 UpdateInteropBuffers();
742 UpdateSolverParameters();
743
744 // execute simulation
745 SmokeAndFireBridge.SubmitInstanceEvent(solverCommandBuffer, CurrentInstanceID,
746 SmokeAndFireBridge.EventID.StepPhysics);
747 Graphics.ExecuteCommandBuffer(solverCommandBuffer);
748
749 // the actual position of the container
750 Vector3 prevPosition = SimulationContainerPosition;
751 SimulationContainerPosition = SmokeAndFireBridge.GetSimulationContainerPosition(CurrentInstanceID);
752 isSimulationContainerPositionChanged = prevPosition != SimulationContainerPosition;
753
754 // update internal time
755 SimulationInternalTime += LastTimestep;
756 SimulationInternalFrame++;
757 }

835 private void RefreshEmitterColorsTexture()
836 {
837 var emitters = Manipulators.FindAll(manip => manip is ZibraParticleEmitter);
838
839 var textureFormat = GraphicsFormat.R8G8B8A8_UNorm;
840 var textureFlags = TextureCreationFlags.None;
841
842 emitters.Sort(new ManipulatorCompare());
843 if (EmittersColorsTexture == null && emitters.Count == 0)
844 {
845 EmittersColorsTexture = new Texture2D(1, 1, textureFormat, textureFlags);
846 EmittersColorsStagingTexture = new Texture2D(1, 1, textureFormat, textureFlags);
847 }
848 else if (EmittersColorsTexture == null || emitters.Count != EmittersColorsTexture.height)
849 {
850 EmittersColorsTexture = new Texture2D(EMITTER_GRADIENT_TEX_WIDTH, Mathf.Max(emitters.Count, 1),
851 textureFormat, textureFlags);
852 EmittersColorsStagingTexture = new Texture2D(EMITTER_GRADIENT_TEX_WIDTH, Mathf.Max(emitters.Count, 1),
853 textureFormat, textureFlags);
854 }
855
856 if (EmittersSpriteTexture == null)
857 {
858 int[] dimensions = new int[] { 1, 1, Mathf.Max(1, emitters.Count) };
859 if (emitters.Find(emitter => (emitter as ZibraParticleEmitter).RenderMode ==
860 ZibraParticleEmitter.RenderingMode.Sprite))
861 {
862 dimensions[0] = dimensions[1] = EMITTER_SPRITE_TEX_SIZE;
863 }
864 EmittersSpriteTexture =
865 new Texture3D(dimensions[0], dimensions[1], dimensions[2], textureFormat, textureFlags);
866 }
867
868 float inv = 1f / (EMITTER_GRADIENT_TEX_WIDTH - 1);
869 for (int y = 0; y < emitters.Count; y++)
870 {
871 var curEmitter = emitters[y] as ZibraParticleEmitter;
872 for (int x = 0; x < EMITTER_GRADIENT_TEX_WIDTH; x++)
873 {
874 var t = x * inv;
875 Color col = curEmitter.ParticleColor.Evaluate(t);
876 col.a = curEmitter.SizeCurve.Evaluate(t) * EMITTER_PARTICLE_SIZE_SCALE;
877 // Can't do SetPixel/Apply on texture passed to native plugin
878 // That will invalidate texture on Metal and crash!
879 EmittersColorsStagingTexture.SetPixel(x, y, col);
880 }
881
882 if (curEmitter.RenderMode == ZibraParticleEmitter.RenderingMode.Sprite)
883 {
884 RenderTexture rt =
885 new RenderTexture(EMITTER_SPRITE_TEX_SIZE, EMITTER_SPRITE_TEX_SIZE, 0, textureFormat);
886 Graphics.Blit(curEmitter.ParticleSprite, rt);
887 int slice = y;
888 Graphics.CopyTexture(rt, 0, EmittersSpriteTexture, slice);
889 }
890 }
891 EmittersColorsStagingTexture.Apply();
892 Graphics.CopyTexture(EmittersColorsStagingTexture, EmittersColorsTexture);
893 }
