148 protected SDFObjectParams GetSDF(SDFObject obj, Manipulator manipulator)
149 {
150 SDFObjectParams sdf = new SDFObjectParams();
151
152 if (obj == null)
153 {
154 throw new Exception("Missing SDF on manipulator");
155 }
156
157 sdf.Rotation = obj.transform.rotation;
158 sdf.Scale = obj.transform.lossyScale;
159 sdf.Position = obj.transform.position;
160
161 sdf.NormalSmooth = 0.01f;
162 sdf.SurfaceValue = 0.0f;
163 SDFObject main = manipulator.GetComponent<SDFObject>();
164 if (main != null)
165 {
166 sdf.SurfaceValue += main.SurfaceDistance;
167 }
168 sdf.SurfaceValue += obj.SurfaceDistance;
169 sdf.DistanceScale = 1.0f;
170 sdf.TotalGroupVolume = 0.0f;
171 sdf.BBoxSize = 0.5f * manipulator.transform.lossyScale;
172
173 if (obj is AnalyticSDF)
174 {
175 AnalyticSDF analyticSDF = obj as AnalyticSDF;
176 sdf.Type = (int)analyticSDF.ChosenSDFType;
177 sdf.DistanceScale = analyticSDF.InvertSDF ? -1.0f : 1.0f;
178 sdf.BBoxSize = analyticSDF.GetBBoxSize();
179 }
180
181 if (obj is NeuralSDF)
182 {
183 NeuralSDF neuralSDF = obj as NeuralSDF;
184 Matrix4x4 transf = obj.transform.localToWorldMatrix * neuralSDF.ObjectRepresentation.ObjectTransform;
185
186 sdf.Rotation = transf.rotation;
187 sdf.Scale = Abs(transf.lossyScale) * (1.0f + 0.1f);
188 sdf.Position = transf.MultiplyPoint(Vector3.zero);
189 sdf.Type = -1;
190 sdf.ObjectID = textureHashMap[neuralSDF.ObjectRepresentation.GetHash()];
191 sdf.EmbeddingTextureBlocks = EmbeddingTextureBlocks;
192 sdf.SDFTextureBlocks = SDFTextureBlocks;
193 sdf.DistanceScale = neuralSDF.InvertSDF ? -1.0f : 1.0f;
194 sdf.BBoxSize = sdf.Scale;
195 }
196
197 sdf.BBoxVolume = sdf.BBoxSize.x * sdf.BBoxSize.y * sdf.BBoxSize.z;
198 return sdf;
199 }

201 protected void AddTexture(NeuralSDF neuralSDF)
202 {
203 ZibraHash128 curHash = neuralSDF.ObjectRepresentation.GetHash();
204
205 if (textureHashMap.ContainsKey(curHash))
206 return;
207
208 SDFTextureSize +=
209 neuralSDF.ObjectRepresentation.GridResolution / NeuralSDFRepresentation.BLOCK_SDF_APPROX_DIMENSION;
210 EmbeddingTextureSize += NeuralSDFRepresentation.EMBEDDING_SIZE *
211 neuralSDF.ObjectRepresentation.EmbeddingResolution /
212 NeuralSDFRepresentation.BLOCK_EMBEDDING_GRID_DIMENSION;
213 neuralSDFs[curHash] = neuralSDF;
214
215 int sdfID = TextureCount;
216 textureHashMap[curHash] = sdfID;
217
218 TextureCount++;
219 }

221 protected void AddTextureData(NeuralSDF neuralSDF)
222 {
223 ZibraHash128 curHash = neuralSDF.ObjectRepresentation.GetHash();
224 int sdfID = textureHashMap[curHash];
225
226 // Embedding texture
227 for (int t = 0; t < NeuralSDFRepresentation.EMBEDDING_SIZE; t++)
228 {
229 int block = sdfID * NeuralSDFRepresentation.EMBEDDING_SIZE + t;
230 Vector3Int blockPos = NeuralSDFRepresentation.BLOCK_EMBEDDING_GRID_DIMENSION *
231 new Vector3Int(block % EmbeddingTextureBlocks,
232 (block / EmbeddingTextureBlocks) % EmbeddingTextureBlocks,
233 block / (EmbeddingTextureBlocks * EmbeddingTextureBlocks));
234 int Size = neuralSDF.ObjectRepresentation.EmbeddingResolution;
235
236 for (int i = 0; i < Size; i++)
237 {
238 for (int j = 0; j < Size; j++)
239 {
240 for (int k = 0; k < Size; k++)
241 {
242 Vector3Int pos = blockPos + new Vector3Int(i, j, k);
243 int id = pos.x + EmbeddingTextureDimension * (pos.y + EmbeddingTextureDimension * pos.z);
244 if (id >= EmbeddingTextureSize)
245 {
246 Debug.LogError(pos);
247 }
248 Embeddings[id] = neuralSDF.ObjectRepresentation.GetEmbedding(i, j, k, t);
249 }
250 }
251 }
252 }
253
254 // SDF approximation texture
255 {
256 int block = sdfID;
257 Vector3Int blockPos =
258 NeuralSDFRepresentation.BLOCK_SDF_APPROX_DIMENSION *
259 new Vector3Int(block % SDFTextureBlocks, (block / SDFTextureBlocks) % SDFTextureBlocks,
260 block / (SDFTextureBlocks * SDFTextureBlocks));
261 int Size = neuralSDF.ObjectRepresentation.GridResolution;
262 for (int i = 0; i < Size; i++)
263 {
264 for (int j = 0; j < Size; j++)
265 {
266 for (int k = 0; k < Size; k++)
267 {
268 Vector3Int pos = blockPos + new Vector3Int(i, j, k);
269 int id = pos.x + SDFTextureDimension * (pos.y + SDFTextureDimension * pos.z);
270 for (int t = 0; t < 2; t++)
271 SDFGrid[2 * id + t] = neuralSDF.ObjectRepresentation.GetSDGrid(i, j, k, t);
272 }
273 }
274 }
275 }

278 protected void CalculateTextureData()
279 {
280 SDFTextureBlocks = (int)Mathf.Ceil(Mathf.Pow(SDFTextureSize, (1.0f / 3.0f)));
281 EmbeddingTextureBlocks = (int)Mathf.Ceil(Mathf.Pow(EmbeddingTextureSize, (1.0f / 3.0f)));
282
283 SDFTextureDimension = NeuralSDFRepresentation.BLOCK_SDF_APPROX_DIMENSION * SDFTextureBlocks;
284 EmbeddingTextureDimension = NeuralSDFRepresentation.BLOCK_EMBEDDING_GRID_DIMENSION * EmbeddingTextureBlocks;
285
286 SDFTextureSize = SDFTextureDimension * SDFTextureDimension * SDFTextureDimension;
287 EmbeddingTextureSize = EmbeddingTextureDimension * EmbeddingTextureDimension * EmbeddingTextureDimension;
288
289 Array.Resize<Color32>(ref Embeddings, EmbeddingTextureSize);
290 Array.Resize<byte>(ref SDFGrid, 2 * SDFTextureSize);
291
292 foreach (var sdf in neuralSDFs.Values)
293 {
294 AddTextureData(sdf);
295 }

303 public void UpdateDynamic(ZibraSmokeAndFire parent, float deltaTime = 0.0f)
304 {
305 int ID = 0;
306 ManipulatorParams.Clear();
307 List<SDFObjectParams> SDFObjectOld = new List<SDFObjectParams>(SDFObjectList);
308 SDFObjectList.Clear();
309 // fill arrays
310
311 foreach (var manipulator in Manipulators)
312 {
313 if (manipulator == null)
314 continue;
315
316 ManipulatorParam manip = new ManipulatorParam();
317
318 manip.Enabled = (manipulator.isActiveAndEnabled && manipulator.gameObject.activeInHierarchy) ? 1 : 0;
319 manip.AdditionalData0 = manipulator.AdditionalData0;
320 manip.AdditionalData1 = manipulator.AdditionalData1;
321
322 SDFObjectParams sdf = GetSDF(manipulator.GetComponent<SDFObject>(), manipulator);
323
324 if (manipulator.GetComponent<SkinnedMeshSDF>() != null)
325 {
326 float TotalVolume = 0.0f;
327 Vector3 averageScale = Vector3.zero;
328 Vector3 averagePosition = Vector3.zero;
329 SkinnedMeshSDF skinnedMeshSDF = manipulator.GetComponent<SkinnedMeshSDF>();
330
331 sdf.Type = -2;
332 sdf.ObjectID = SDFObjectList.Count;
333 sdf.SDFTextureBlocks = skinnedMeshSDF.BoneSDFList.Count;
334
335 foreach (var bone in skinnedMeshSDF.BoneSDFList)
336 {
337 SDFObjectParams boneSDF = GetSDF(bone, manipulator);
338 TotalVolume += boneSDF.BBoxVolume;
339 averageScale += boneSDF.Scale;
340 averagePosition += boneSDF.Position;
341
342 if (SDFObjectOld.Count > 0)
343 {
344 boneSDF.PrevPosition = SDFObjectOld[SDFObjectList.Count].Position;
345 boneSDF.PrevScale = SDFObjectOld[SDFObjectList.Count].Scale;
346 boneSDF.PrevRotation = SDFObjectOld[SDFObjectList.Count].Rotation;
347 }
348 else
349 {
350 boneSDF.PrevPosition = boneSDF.Position;
351 boneSDF.PrevScale = boneSDF.Scale;
352 boneSDF.PrevRotation = boneSDF.Rotation;
353 }
354
355 SDFObjectList.Add(boneSDF);
356 }
357
358 sdf.Position = averagePosition / skinnedMeshSDF.BoneSDFList.Count;
359 sdf.Scale = averageScale / skinnedMeshSDF.BoneSDFList.Count;
360 sdf.TotalGroupVolume = TotalVolume;
361 }
362
363 if (manipulator is ZibraSmokeAndFireEmitter)
364 {
365 ZibraSmokeAndFireEmitter emitter = manipulator as ZibraSmokeAndFireEmitter;
366
367 manip.AdditionalData0.x = 0.0f;
368
369 Vector3 normalizedColor =
370 new Vector3(emitter.SmokeColor.r, emitter.SmokeColor.g, emitter.SmokeColor.b);
371 if (normalizedColor.sqrMagnitude < 0.001f)
372 {
373 normalizedColor = new Vector3(1.0f, 1.0f, 1.0f);
374 }
375 normalizedColor.Normalize();
376 Vector3 effectiveColor = normalizedColor * emitter.SmokeDensity;
377
378 if (parent.ActiveSimulationMode == ZibraSmokeAndFire.SimulationMode.Fire)
379 {
380 manip.AdditionalData1 =
381 new Vector4(emitter.SmokeDensity, emitter.EmitterFuel, emitter.EmitterTemperature,
382 emitter.UseObjectVelocity ? 1.0f : 0.0f);
383 }
384 else
385 {
386 manip.AdditionalData1 = new Vector4(effectiveColor[0], effectiveColor[1], effectiveColor[2],
387 emitter.UseObjectVelocity ? 1.0f : 0.0f);
388 }
389 }
390
391 if (manipulator is ZibraSmokeAndFireVoid)
392 {
393 ZibraSmokeAndFireVoid voidmanip = manipulator as ZibraSmokeAndFireVoid;
394 manip.AdditionalData0.x = voidmanip.ColorDecay;
395 manip.AdditionalData0.y = voidmanip.VelocityDecay;
396 manip.AdditionalData0.z = voidmanip.Pressure;
397 }
398
399 manip.SDFObjectID = SDFObjectList.Count;
400
401 if (SDFObjectOld.Count > 0)
402 {
403 sdf.PrevPosition = SDFObjectOld[manip.SDFObjectID].Position;
404 sdf.PrevScale = SDFObjectOld[manip.SDFObjectID].Scale;
405 sdf.PrevRotation = SDFObjectOld[manip.SDFObjectID].Rotation;
406 }
407 else
408 {
409 sdf.PrevPosition = sdf.Position;
410 sdf.PrevScale = sdf.Scale;
411 sdf.PrevRotation = sdf.Rotation;
412 }
413 SDFObjectList.Add(sdf);
414 ManipulatorParams.Add(manip);
415 ID++;
416 }
417
418 Elements = Manipulators.Count;
419 }

438 public void UpdateStatistics(Int32[] data, List<Manipulator> curManipulators,
439 DataStructures.ZibraSmokeAndFireSolverParameters solverParameters,
440 DataStructures.ZibraSmokeAndFireMaterialParameters materialParameters)

466 break;
467 }
468#if UNITY_EDITOR
469 manipulator.NotifyChange();
470#endif
471
472 id++;
473 }

480 public void UpdateConst(List<Manipulator> curManipulators)
481 {
482 Manipulators = new List<Manipulator>();
483
484 // add all colliders to the manipulator list
485 foreach (var manipulator in curManipulators)
486 {
487 if (manipulator == null)
488 continue;
489
490 var sdf = manipulator.GetComponent<SDFObject>();
491 if (sdf == null)
492 {
493 Debug.LogWarning("Manipulator " + manipulator.gameObject.name + " missing sdf and is disabled.");
494 continue;
495 }
496
497 NeuralSDF neuralSDF = sdf as NeuralSDF;
498 if (neuralSDF != null && !neuralSDF.ObjectRepresentation.HasRepresentationV3)
499 {
500 Debug.LogWarning("NeuralSDF in " + manipulator.gameObject.name +
501 " was not generated. Manipulator is disabled.");
502 continue;
503 }
504
505 TerrainSDF terrainSDF = sdf as TerrainSDF;
506 if (terrainSDF != null)
507 {
508 Debug.LogWarning("TerrainSDF was used in " + manipulator.gameObject.name +
509 " which is not supported. TerrainSDF only supported in colliders. " +
510 "Manipulator is disabled.");
511 continue;
512 }
513
514 if (sdf is SkinnedMeshSDF)
515 {
516 SkinnedMeshSDF skinnedMeshSDF = sdf as SkinnedMeshSDF;
517 if (!skinnedMeshSDF.HasRepresentation())
518 {
519 Debug.LogWarning("SkinnedMeshSDF in " + manipulator.gameObject.name +
520 " was not generated. Manipulator is disabled.");
521 continue;
522 }
523 }
524
525 if (sdf is SkinnedMeshSDF)
526 {
527 SkinnedMeshSDF skinnedMeshSDF = sdf as SkinnedMeshSDF;
528 if (!skinnedMeshSDF.HasRepresentation())
529 {
530 Debug.LogWarning("SkinnedMeshSDF in " + manipulator.gameObject.name +
531 " was not generated. Manipulator is disabled.");
532 continue;
533 }
534 }
535
536 Manipulators.Add(manipulator);
537 }
538
539 // first sort the manipulators
540 Manipulators.Sort(new ManipulatorCompare());
541
542 int[] TypeIndex = new int[(int)Manipulator.ManipulatorType.TypeNum * 2];
543 foreach (Manipulator.ManipulatorType curManipulatorType in Enum.GetValues(
544 typeof(Manipulator.ManipulatorType)))
545 {
546 if (curManipulatorType == Manipulator.ManipulatorType.None ||
547 curManipulatorType == Manipulator.ManipulatorType.TypeNum)
548 {
549 continue;
550 }
551 Predicate<Manipulator> FindType = delegate(Manipulator curManipulator)
552 {
553 return curManipulator.GetManipulatorType() == curManipulatorType;
554 };
555 int firstIndex = Manipulators.FindIndex(FindType);
556 int quantity = Manipulators.FindAll(FindType).Count;
557
558 int curTypeIndex = (int)curManipulatorType * 2;
559 TypeIndex[curTypeIndex] = firstIndex;
560 TypeIndex[curTypeIndex + 1] = firstIndex + quantity;
561 }
562
563 indices.EmitterIndexBegin = TypeIndex[2 * (int)Manipulator.ManipulatorType.Emitter];
564 indices.EmitterIndexEnd = TypeIndex[2 * (int)Manipulator.ManipulatorType.Emitter + 1];
565 indices.VoidIndexBegin = TypeIndex[2 * (int)Manipulator.ManipulatorType.Void];
566 indices.VoidIndexEnd = TypeIndex[2 * (int)Manipulator.ManipulatorType.Void + 1];
567 indices.ForceFieldIndexBegin = TypeIndex[2 * (int)Manipulator.ManipulatorType.ForceField];
568 indices.ForceFieldIndexEnd = TypeIndex[2 * (int)Manipulator.ManipulatorType.ForceField + 1];
569 indices.AnalyticColliderIndexBegin = TypeIndex[2 * (int)Manipulator.ManipulatorType.AnalyticCollider];
570 indices.AnalyticColliderIndexEnd = TypeIndex[2 * (int)Manipulator.ManipulatorType.AnalyticCollider + 1];
571 indices.NeuralColliderIndexBegin = TypeIndex[2 * (int)Manipulator.ManipulatorType.NeuralCollider];
572 indices.NeuralColliderIndexEnd = TypeIndex[2 * (int)Manipulator.ManipulatorType.NeuralCollider + 1];
573 indices.GroupColliderIndexBegin = TypeIndex[2 * (int)Manipulator.ManipulatorType.GroupCollider];
574 indices.GroupColliderIndexEnd = TypeIndex[2 * (int)Manipulator.ManipulatorType.GroupCollider + 1];
575 indices.DetectorIndexBegin = TypeIndex[2 * (int)Manipulator.ManipulatorType.Detector];
576 indices.DetectorIndexEnd = TypeIndex[2 * (int)Manipulator.ManipulatorType.Detector + 1];
577 indices.SpeciesModifierIndexBegin = TypeIndex[2 * (int)Manipulator.ManipulatorType.SpeciesModifier];
578 indices.SpeciesModifierIndexEnd = TypeIndex[2 * (int)Manipulator.ManipulatorType.SpeciesModifier + 1];
579 indices.EffectParticleEmitterBegin = TypeIndex[2 * (int)Manipulator.ManipulatorType.EffectParticleEmitter];
580 indices.EffectParticleEmitterEnd =
581 TypeIndex[2 * (int)Manipulator.ManipulatorType.EffectParticleEmitter + 1];
582 indices.TextureEmitterIndexBegin = TypeIndex[2 * (int)Manipulator.ManipulatorType.TextureEmitter];
583 indices.TextureEmitterIndexEnd = TypeIndex[2 * (int)Manipulator.ManipulatorType.TextureEmitter + 1];
584
585 int enabledForceFieldsCount = indices.ForceFieldIndexEnd - indices.ForceFieldIndexBegin;
586 if (enabledForceFieldsCount > 0)
587 {
588 var forceFieldsList = Manipulators.GetRange(indices.ForceFieldIndexBegin, enabledForceFieldsCount);
589 { // Analytic force fields
590 Predicate<Manipulator> FindAnalyticSDF = delegate(Manipulator curManipulator)
591 {
592 return curManipulator.GetComponent<SDFObject>().GetSDFType() ==
593 SDFObject.SDFObjectType.Analytic;
594 };
595 int firstAnalytycFFIndex = forceFieldsList.FindIndex(FindAnalyticSDF);
596 int ffQuantity = forceFieldsList.FindAll(FindAnalyticSDF).Count;
597
598 indices.AnalyticForceFieldIndexBegin = indices.ForceFieldIndexBegin + firstAnalytycFFIndex;
599 indices.AnalyticForceFieldIndexEnd =
600 indices.ForceFieldIndexBegin + firstAnalytycFFIndex + ffQuantity;
601 }
602
603 { // Neural force fields
604 Predicate<Manipulator> FindNeuralSDF = delegate(Manipulator curManipulator)
605 {
606 return curManipulator.GetComponent<SDFObject>().GetSDFType() == SDFObject.SDFObjectType.Neural;
607 };
608 int firstNeuralFFIndex = forceFieldsList.FindIndex(FindNeuralSDF);
609 int ffQuantity = forceFieldsList.FindAll(FindNeuralSDF).Count;
610
611 indices.NeuralForceFieldIndexBegin = indices.ForceFieldIndexBegin + firstNeuralFFIndex;
612 indices.NeuralForceFieldIndexEnd = indices.ForceFieldIndexBegin + firstNeuralFFIndex + ffQuantity;
613 }
614
615 { // Group force fields
616 Predicate<Manipulator> FindGroupSDF = delegate(Manipulator curManipulator)
617 {
618 return curManipulator.GetComponent<SDFObject>().GetSDFType() == SDFObject.SDFObjectType.Group;
619 };
620 int firstGroupFFIndex = forceFieldsList.FindIndex(FindGroupSDF);
621 int ffQuantity = forceFieldsList.FindAll(FindGroupSDF).Count;
622
623 indices.GroupForceFieldIndexBegin = indices.ForceFieldIndexBegin + firstGroupFFIndex;
624 indices.GroupForceFieldIndexEnd = indices.ForceFieldIndexBegin + firstGroupFFIndex + ffQuantity;
625 }
626 }
627 else
628 {
629 indices.AnalyticForceFieldIndexBegin = 0;
630 indices.AnalyticForceFieldIndexEnd = 0;
631 indices.NeuralForceFieldIndexBegin = 0;
632 indices.NeuralForceFieldIndexEnd = 0;
633 indices.GroupForceFieldIndexBegin = 0;
634 indices.GroupForceFieldIndexEnd = 0;
635 }
636
637 ConstDataID.Clear();
638
639 SDFTextureSize = 0;
640 EmbeddingTextureSize = 0;
641 textureHashMap.Clear();
642 TextureCount = 0;
643 foreach (var manipulator in Manipulators)
644 {
645 if (manipulator == null)
646 continue;
647
648 if (manipulator.GetComponent<NeuralSDF>() != null)
649 {
650 AddTexture(manipulator.GetComponent<NeuralSDF>());
651 }
652
653 if (manipulator.GetComponent<SkinnedMeshSDF>() != null)
654 {
655 SkinnedMeshSDF skinnedMeshSDF = manipulator.GetComponent<SkinnedMeshSDF>();
656
657 foreach (var bone in skinnedMeshSDF.BoneSDFList)
658 {
659 if (bone is NeuralSDF neuralBone)
660 AddTexture(neuralBone);
661 }
662 }
663
