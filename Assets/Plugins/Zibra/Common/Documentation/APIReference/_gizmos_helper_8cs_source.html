47 public static void DrawWireTorus(Vector3 pos, Quaternion rot, float radiusSmall, float radiusLarge)
48 {
49 Matrix4x4 angleMatrix = Matrix4x4.TRS(pos, rot, Handles.matrix.lossyScale);
50 using (new Handles.DrawingScope(angleMatrix))
51 {
52 const float step = 40.0f;
53
54 // small circles
55 for (float ang = 0.0f; ang < 360.0f; ang += step)
56 {
57 float radians = Mathf.Deg2Rad * ang;
58 Vector3 direction = Vector3.left * Mathf.Sin(radians) + Vector3.forward * Mathf.Cos(radians);
59 Vector3 normal = Vector3.left * Mathf.Cos(radians) - Vector3.forward * Mathf.Sin(radians);
60 Handles.DrawWireDisc(direction * radiusLarge, normal, radiusSmall);
61 }
62
63 // large circles
64 for (float ang = 0.0f; ang < 360.0f; ang += step)
65 {
66 float radians = Mathf.Deg2Rad * ang;
67 float radius = radiusLarge + radiusSmall * Mathf.Sin(radians);
68 Vector3 center = radiusSmall * Vector3.up * Mathf.Cos(radians);
69 Handles.DrawWireDisc(center, Vector3.up, radius);
70 }
71 }

89 private static float Phi = 0.5f * (Mathf.Sqrt(5.0f) + 1.0f);
90
91 // i-th point of n uniformly distibuted fibonacci points on a sphere
92 public static Vector3 FibonacciSpherePoint(int i, int n)
93 {
94 Vector2 pt = new Vector2(2.0f * Mathf.PI * ((i / Phi) % 1.0f), Mathf.Acos(1.0f - (2.0f * i + 1.0f) / n));
95 return new Vector3(Mathf.Cos(pt.x) * Mathf.Sin(pt.y), Mathf.Sin(pt.x) * Mathf.Sin(pt.y), Mathf.Cos(pt.y));
96 }
