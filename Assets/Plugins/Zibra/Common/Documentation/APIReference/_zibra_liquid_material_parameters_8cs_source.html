604 [ExecuteInEditMode]
605 private void Awake()
606 {
607 // If Material Parameters is in old format we need to parse old parameters and come up with equivalent new
608 // ones
609#if UNITY_EDITOR
610 bool updated = false;
611#endif
612
613 if (ObjectVersion == 1)
614 {
615 Roughness = 1 - SmoothnessOld;
616
617 ObjectVersion = 2;
618#if UNITY_EDITOR
619 updated = true;
620#endif
621 }
622
623 if (ObjectVersion == 2)
624 {
625 Solver.ZibraLiquid instance = GetComponent<Solver.ZibraLiquid>();
626
627 // if not a newly created liquid instance
628 //(material parameters are created before liquid)
629 if (instance != null)
630 {
631 const float TotalScale = 0.33f;
632 float SimulationScale =
633 TotalScale * (instance.ContainerSize.x + instance.ContainerSize.y + instance.ContainerSize.z);
634
635 ScatteringAmount *= SimulationScale;
636 AbsorptionAmount *= SimulationScale;
637 }
638
639 ObjectVersion = 3;
640
641#if UNITY_EDITOR
642 updated = true;
643#endif
644 }
645
646#if UNITY_EDITOR
647 if (updated)
648 {
649 // Can't mark object dirty in Awake, since scene is not fully loaded yet
650 UnityEditor.SceneManagement.EditorSceneManager.sceneOpened += OnSceneOpened;
651 }
652#endif
653 }
