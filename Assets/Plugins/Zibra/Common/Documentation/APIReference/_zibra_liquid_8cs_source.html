1409 public void InitializeSimulation()
1410 {
1411 if (Initialized)
1412 {
1413 return;
1414 }
1415
1416 try
1417 {
1418#if UNITY_IOS
1419 if (SystemInfo.graphicsDeviceName == "Apple iOS simulator GPU")
1420 {
1421 throw new Exception("Zibra Liquids doesn't support iOS simulator. " + "Liquid was disabled.");
1422 }
1423#endif
1424
1425#if !ZIBRA_EFFECTS_NO_LICENSE_CHECK && UNITY_EDITOR
1426 if (!ServerAuthManager.GetInstance().IsLicenseVerified(ServerAuthManager.Effect.Liquids))
1427 {
1428 string errorMessage =
1429 "License wasn't verified. " +
1430 ServerAuthManager.GetInstance().GetErrorMessage(ServerAuthManager.Effect.Liquids) +
1431 " Liquid won't run in editor.";
1432 throw new Exception(errorMessage);
1433 }
1434#endif
1435
1436 // Copy Rendering Mode to internal variable to prevent user from modifying it during liquid lifetime
1437 ActiveRenderingMode = CurrentRenderingMode;
1438
1439#if UNITY_PIPELINE_HDRP
1440 if (RenderPipelineDetector.GetRenderPipelineType() == RenderPipelineDetector.RenderPipeline.HDRP)
1441 {
1442 bool missingRequiredParameter = false;
1443
1444 if (CustomLightHDRP == null && ActiveRenderingMode != RenderingMode.UnityRender)
1445 {
1446 Debug.LogError("No Custom Light set in Zibra Liquid.");
1447 missingRequiredParameter = true;
1448 }
1449
1450 if (ReflectionProbeHDRP == null && ActiveRenderingMode != RenderingMode.UnityRender)
1451 {
1452 Debug.LogError("No reflection probe added to Zibra Liquid.");
1453 missingRequiredParameter = true;
1454 }
1455
1456 if (missingRequiredParameter)
1457 {
1458 throw new Exception("Liquid creation failed due to missing parameter.");
1459 }
1460 }
1461#endif
1462
1463 ValidateColliders();
1464 ValidateManipulators();
1465
1466 if (InitialState == ZibraLiquid.InitialStateType.NoParticles || BakedInitialStateAsset == null)
1467 {
1468 bool haveEmitter = false;
1469 foreach (var manipulator in Manipulators)
1470 {
1471 if (manipulator.GetManipulatorType() == Manipulator.ManipulatorType.Emitter &&
1472 manipulator.GetComponent<SDFObject>() != null)
1473 {
1474 haveEmitter = true;
1475 break;
1476 }
1477 }
1478
1479 if (!haveEmitter)
1480 {
1481 throw new Exception(
1482 "Liquid creation failed. Liquid have neither initial state nor emitters, or all emitters missing SDF component.");
1483 }
1484 }
1485
1486 Camera.onPreRender += RenderCallBackWrapper;
1487
1488 SolverCommandBuffer = new CommandBuffer { name = "ZibraLiquid.Solver" };
1489
1490 CurrentInstanceID = NextInstanceId++;
1491
1492 ForceCloseCommandEncoder(SolverCommandBuffer);
1493 LiquidBridge.SubmitInstanceEvent(SolverCommandBuffer, CurrentInstanceID,
1494 LiquidBridge.EventID.CreateFluidInstance);
1495 Graphics.ExecuteCommandBuffer(SolverCommandBuffer);
1496 SolverCommandBuffer.Clear();
1497
1498 InitializeParticles();
1499
1500 var initializeGPUReadbackParamsBridgeParams = new InitializeGPUReadbackParams();
1501 UInt32 manipSize = (UInt32)ManipulatorManager.Elements * STATISTICS_PER_MANIPULATOR * sizeof(Int32);
1502 initializeGPUReadbackParamsBridgeParams.readbackBufferSize = sizeof(Int32) + manipSize;
1503 switch (SystemInfo.graphicsDeviceType)
1504 {
1505 case GraphicsDeviceType.Direct3D11:
1506 case GraphicsDeviceType.XboxOne:
1507 case GraphicsDeviceType.Switch:
1508#if UNITY_2020_3_OR_NEWER
1509 case GraphicsDeviceType.Direct3D12:
1510 case GraphicsDeviceType.XboxOneD3D12:
1511#endif
1512 initializeGPUReadbackParamsBridgeParams.maxFramesInFlight = QualitySettings.maxQueuedFrames + 1;
1513 break;
1514 default:
1515 initializeGPUReadbackParamsBridgeParams.maxFramesInFlight = (int)MaxFramesInFlight;
1516 break;
1517 }
1518
1519 IntPtr nativeCreateInstanceBridgeParams =
1520 Marshal.AllocHGlobal(Marshal.SizeOf(initializeGPUReadbackParamsBridgeParams));
1521 Marshal.StructureToPtr(initializeGPUReadbackParamsBridgeParams, nativeCreateInstanceBridgeParams, true);
1522
1523 SolverCommandBuffer.Clear();
1524 LiquidBridge.SubmitInstanceEvent(SolverCommandBuffer, CurrentInstanceID,
1525 LiquidBridge.EventID.InitializeGpuReadback,
1526 nativeCreateInstanceBridgeParams);
1527 Graphics.ExecuteCommandBuffer(SolverCommandBuffer);
1528
1529 ToFreeOnExit.Add(nativeCreateInstanceBridgeParams);
1530
1531 InitializeSolver();
1532
1533 Initialized = true;
1534
1535#if UNITY_EDITOR
1536 ZibraLiquidAnalytics.TrackSimulationInitialization(this);
1537#endif
1538 }
1539 catch (Exception)
1540 {
1541 ClearRendering();
1542 ClearSolver();
1543 throw;
1544 }
1545 }
