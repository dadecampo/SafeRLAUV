165 var embeds = CurrentRepresentationV3.embeds.StringToBytes();
166 VoxelInfo.grid = CurrentRepresentationV3.sd_grid.StringToBytes();

172
173 Array.Resize<Color32>(ref VoxelInfo.embeds, embeddingSize * EMBEDDING_SIZE);
174
175 for (int i = 0; i < embeddingResolution; i++)

177 for (int j = 0; j < embeddingResolution; j++)

179 for (int k = 0; k < embeddingResolution; k++)

181 for (int t = 0; t < EMBEDDING_SIZE; t++)
182 {
183 int id0 = i + t * embeddingResolution +
184 EMBEDDING_SIZE * embeddingResolution * (j + k * embeddingResolution);
185 int id1 = t + (i + embeddingResolution * (j + k * embeddingResolution)) * EMBEDDING_SIZE;
186 Color32 embeddings = new Color32(embeds[PACKING * id1 + 0], embeds[PACKING * id1 + 1],
187 embeds[PACKING * id1 + 2], embeds[PACKING * id1 + 3]);
