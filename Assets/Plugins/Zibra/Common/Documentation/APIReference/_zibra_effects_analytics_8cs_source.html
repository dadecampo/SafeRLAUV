17 const int REPORT_PERIOD = 12 * 60 * 60; // 12 hours
18 const string ENGINE = "Unity";
19 const int TRACKING_VERSION = 2;
20 const int FRAME_UPDATE_PERIOD = 15 * 60 * 30; // 15 minutes if FPS = 30
21 const int SESSION_TIMEOUT = 15 * 60; // 15 minutes
22
23 public static void TrackBuiltPlatform(string builtPlatform)
24 {
25 EditorPrefs.SetBool($"ZibraEffectsTracking_Common_Built{builtPlatform}", true);
26 }
27
28 public static void TrackConfiguration(string effect)
29 {
30 EditorPrefs.SetBool("ZibraEffectsTracking_" + effect + "_Configured", true);
31
32 string sessionKeyName = "ZibraEffectsTracking_" + effect + "_LastActivity";
33 string prefsKeyName = "ZibraEffectsTracking_" + effect + "_SessionTime";
34 int currentTime = GetCurrentTimeAsUnixTimestamp();
35 int lastActivity = SessionState.GetInt(sessionKeyName, currentTime);
36 if (currentTime - lastActivity < SESSION_TIMEOUT)
37 {
38 int oldSessionTime = EditorPrefs.GetInt(prefsKeyName, 0);
39 EditorPrefs.SetInt(prefsKeyName, oldSessionTime + currentTime - lastActivity);
40 }
41 SessionState.SetInt(sessionKeyName, currentTime);
42 }
43
44 public static void CheckSendPeriod()
45 {
46 int dateNow = GetCurrentTimeAsUnixTimestamp();
47
48 if (!EditorPrefs.HasKey("ZibraEffectsTracking_Common_LastSentDate"))
49 {
50 EditorPrefs.SetInt("ZibraEffectsTracking_Common_LastSentDate", dateNow);
51 return;
52 }
53
54 int lastSentDate = EditorPrefs.GetInt("ZibraEffectsTracking_Common_LastSentDate");
55
56 if (dateNow - lastSentDate > REPORT_PERIOD)
57 {
58 ZibraEffectsAnalyticsStruct data = GetAnalyticsData();
59 string jsonData = JsonUtility.ToJson(data);
60 ZibraEffectsAnalyticsSender.SendAnalyticsData(jsonData);
61 }
62 }
63
64 static int FrameCount = FRAME_UPDATE_PERIOD;
65
66 private static void Update()
67 {
68 if (FrameCount == FRAME_UPDATE_PERIOD)
69 {
70 FrameCount = 0;
71 CheckSendPeriod();
72 }
73 else
74 {
75 FrameCount++;
76 }
77 }

88 EditorApplication.update += Update;
89 }
