143 protected SDFObjectParams GetSDF(SDFObject obj, Manipulator manipulator)
144 {
145 SDFObjectParams sdf = new SDFObjectParams();
146
147 if (obj == null)
148 {
149 throw new Exception("Missing SDF on manipulator");
150 }
151
152 sdf.Rotation = manipulator.transform.rotation;
153 sdf.Scale = manipulator.transform.lossyScale;
154 sdf.Position = manipulator.transform.position;
155 sdf.BBoxSize = 2.0f * sdf.Scale;
156
157 sdf.NormalSmooth = 0.01f;
158 sdf.Velocity = Vector3.zero;
159 sdf.SurfaceValue = 0.0f;
160 SDFObject main = manipulator.GetComponent<SDFObject>();
161 if (main != null)
162 {
163 sdf.SurfaceValue += main.SurfaceDistance;
164 }
165 sdf.SurfaceValue += obj.SurfaceDistance;
166 sdf.DistanceScale = 1.0f;
167 sdf.AnguralVelocity = Vector3.zero;
168 sdf.TotalGroupVolume = 0.0f;
169 sdf.BBoxSize = 0.5f * manipulator.transform.lossyScale;
170
171 if (obj is AnalyticSDF)
172 {
173 AnalyticSDF analyticSDF = obj as AnalyticSDF;
174 sdf.Type = (int)analyticSDF.ChosenSDFType;
175 sdf.DistanceScale = analyticSDF.InvertSDF ? -1.0f : 1.0f;
176 sdf.BBoxSize = analyticSDF.GetBBoxSize();
177 }
178
179 if (obj is NeuralSDF)
180 {
181 NeuralSDF neuralSDF = obj as NeuralSDF;
182 Matrix4x4 transf = obj.transform.localToWorldMatrix * neuralSDF.ObjectRepresentation.ObjectTransform;
183
184 sdf.Rotation = transf.rotation;
185 sdf.Scale = Abs(transf.lossyScale);
186 sdf.Position = transf.MultiplyPoint(Vector3.zero);
187 sdf.Type = (int)SDFObject.SDFObjectType.Neural;
188 sdf.ObjectID = TextureHashMap[neuralSDF.ObjectRepresentation.GetHash()];
189 sdf.EmbeddingTextureBlocks = EmbeddingTextureBlocks;
190 sdf.SDFTextureBlocks = SDFTextureBlocks;
191 sdf.DistanceScale = neuralSDF.InvertSDF ? -1.0f : 1.0f;
192 sdf.BBoxSize = sdf.Scale;
193 }
194
195 sdf.BBoxVolume = sdf.BBoxSize.x * sdf.BBoxSize.y * sdf.BBoxSize.z;
196 return sdf;
197 }

199 protected void AddTexture(NeuralSDF neuralSDF)
200 {
201 ZibraHash128 curHash = neuralSDF.ObjectRepresentation.GetHash();
202
203 if (TextureHashMap.ContainsKey(curHash))
204 return;
205
206 SDFTextureSize +=
207 neuralSDF.ObjectRepresentation.GridResolution / NeuralSDFRepresentation.BLOCK_SDF_APPROX_DIMENSION;
208 EmbeddingTextureSize += NeuralSDFRepresentation.EMBEDDING_SIZE *
209 neuralSDF.ObjectRepresentation.EmbeddingResolution /
210 NeuralSDFRepresentation.BLOCK_EMBEDDING_GRID_DIMENSION;
211 NeuralSDFs[curHash] = neuralSDF;
212
213 int sdfID = TextureCount;
214 TextureHashMap[curHash] = sdfID;
215
216 TextureCount++;
217 }

219 protected void AddTextureData(NeuralSDF neuralSDF)
220 {
221 ZibraHash128 curHash = neuralSDF.ObjectRepresentation.GetHash();
222 int sdfID = TextureHashMap[curHash];
223
224 // Embedding texture
225 for (int t = 0; t < NeuralSDFRepresentation.EMBEDDING_SIZE; t++)
226 {
227 int block = sdfID * NeuralSDFRepresentation.EMBEDDING_SIZE + t;
228 Vector3Int blockPos = NeuralSDFRepresentation.BLOCK_EMBEDDING_GRID_DIMENSION *
229 new Vector3Int(block % EmbeddingTextureBlocks,
230 (block / EmbeddingTextureBlocks) % EmbeddingTextureBlocks,
231 block / (EmbeddingTextureBlocks * EmbeddingTextureBlocks));
232 int Size = neuralSDF.ObjectRepresentation.EmbeddingResolution;
233
234 for (int i = 0; i < Size; i++)
235 {
236 for (int j = 0; j < Size; j++)
237 {
238 for (int k = 0; k < Size; k++)
239 {
240 Vector3Int pos = blockPos + new Vector3Int(i, j, k);
241 int id = pos.x + EmbeddingTextureDimension * (pos.y + EmbeddingTextureDimension * pos.z);
242 if (id >= EmbeddingTextureSize)
243 {
244 Debug.LogError(pos);
245 }
246 Embeddings[id] = neuralSDF.ObjectRepresentation.GetEmbedding(i, j, k, t);
247 }
248 }
249 }
250 }
251
252 // SDF approximation texture
253 {
254 int block = sdfID;
255 Vector3Int blockPos =
256 NeuralSDFRepresentation.BLOCK_SDF_APPROX_DIMENSION *
257 new Vector3Int(block % SDFTextureBlocks, (block / SDFTextureBlocks) % SDFTextureBlocks,
258 block / (SDFTextureBlocks * SDFTextureBlocks));
259 int Size = neuralSDF.ObjectRepresentation.GridResolution;
260 for (int i = 0; i < Size; i++)
261 {
262 for (int j = 0; j < Size; j++)
263 {
264 for (int k = 0; k < Size; k++)
265 {
266 Vector3Int pos = blockPos + new Vector3Int(i, j, k);
267 int id = pos.x + SDFTextureDimension * (pos.y + SDFTextureDimension * pos.z);
268 for (int t = 0; t < 2; t++)
269 SDFGrid[2 * id + t] = neuralSDF.ObjectRepresentation.GetSDGrid(i, j, k, t);
270 }
271 }
272 }
273 }

276 protected void CalculateTextureData()
277 {
278 SDFTextureBlocks = (int)Mathf.Ceil(Mathf.Pow(SDFTextureSize, (1.0f / 3.0f)));
279 EmbeddingTextureBlocks = (int)Mathf.Ceil(Mathf.Pow(EmbeddingTextureSize, (1.0f / 3.0f)));
280
281 SDFTextureDimension = NeuralSDFRepresentation.BLOCK_SDF_APPROX_DIMENSION * SDFTextureBlocks;
282 EmbeddingTextureDimension = NeuralSDFRepresentation.BLOCK_EMBEDDING_GRID_DIMENSION * EmbeddingTextureBlocks;
283
284 SDFTextureSize = SDFTextureDimension * SDFTextureDimension * SDFTextureDimension;
285 EmbeddingTextureSize = EmbeddingTextureDimension * EmbeddingTextureDimension * EmbeddingTextureDimension;
286
287 Array.Resize<Color32>(ref Embeddings, EmbeddingTextureSize);
288 Array.Resize<byte>(ref SDFGrid, 2 * SDFTextureSize);
289
290 foreach (var sdf in NeuralSDFs.Values)
291 {
292 AddTextureData(sdf);
293 }
294 }

296 protected void UpdateHeightmap(CommandBuffer commandBuffer, RenderTexture heightmaps, Terrain terrain,
297 ref int heightmapID, ref SDFObjectParams sdf)
298 {
299 if (terrain != null)
300 {
301 TerrainData terrainData = terrain.terrainData;
302 RenderTexture heightmap = terrainData.heightmapTexture;
303 bool singleChannel = true;
304 switch (heightmap.format)
305 {
306 case RenderTextureFormat.RFloat:
307 case RenderTextureFormat.R8:
308 case RenderTextureFormat.R16:
309 case RenderTextureFormat.RHalf:
310 case RenderTextureFormat.RInt:
311 singleChannel = true;
312 break;
313 default:
314 singleChannel = false;
315 break;
316 }
317 Vector2 chunkID = new Vector2(heightmapID % HeightmapCountSqrt, heightmapID / HeightmapCountSqrt);
318 Vector4 scaling = new Vector4(chunkID.x, chunkID.y, HeightmapCountSqrt, HeightmapCountSqrt);
319 commandBuffer.SetGlobalVector("_HeightmapScaling", scaling);
320 commandBuffer.SetGlobalInt("_ComponentToBlit", singleChannel ? 0 : 1);
321 commandBuffer.Blit(heightmap, heightmaps, HeightmapBlit);
322 sdf.Type = (int)SDFObject.SDFObjectType.Heightmap;
323 sdf.ObjectID = heightmapID;
324 sdf.SDFTextureBlocks = HeightmapCountSqrt;
325 sdf.Rotation = Quaternion.identity;
326 sdf.Scale = terrainData.heightmapScale * terrainData.heightmapResolution;
327 // heightmap is from 0 to 1, but the scales of the objects we have are from -1 to 1, so we need to
328 // multiply by 2.0f
329 sdf.Scale.y = terrainData.heightmapScale.y * 2.0f;
330 heightmapID++;
331 }

344 public void UpdateDynamic(CommandBuffer commandBuffer, ZibraLiquid parent, float deltaTime = 0.0f)
345 {
346 int ID = 0;
347 int heightmapID = 0;
348 ManipulatorParams.Clear();
349 SDFObjectList.Clear();
350 // fill arrays
351
352 foreach (var manipulator in Manipulators)
353 {
354 if (manipulator == null)
355 continue;
356
357 ManipulatorParam manip = new ManipulatorParam();
358
359 manip.Enabled = (manipulator.isActiveAndEnabled && manipulator.gameObject.activeInHierarchy) ? 1 : 0;
360 manip.AdditionalData0 = manipulator.AdditionalData0;
361 manip.AdditionalData1 = manipulator.AdditionalData1;
362
363 SDFObjectParams sdf = GetSDF(manipulator.GetComponent<SDFObject>(), manipulator);
364
365 SDFObject sdfobj = manipulator.GetComponent<SDFObject>();
366
367 if (sdfobj is TerrainSDF)
368 {
369 UpdateHeightmap(commandBuffer, parent.HeightmapTexture, manipulator.GetComponent<Terrain>(),
370 ref heightmapID, ref sdf);
371 }
372
373 if (sdfobj is SkinnedMeshSDF)
374 {
375 float TotalVolume = 0.0f;
376 Vector3 averageScale = Vector3.zero;
377 Vector3 averagePosition = Vector3.zero;
378 SkinnedMeshSDF skinnedMeshSDF = manipulator.GetComponent<SkinnedMeshSDF>();
379
380 sdf.Type = (int)SDFObject.SDFObjectType.Group;
381 sdf.ObjectID = SDFObjectList.Count;
382 sdf.SDFTextureBlocks = skinnedMeshSDF.BoneSDFList.Count;
383
384 foreach (var bone in skinnedMeshSDF.BoneSDFList)
385 {
386 SDFObjectParams boneSDF = GetSDF(bone, manipulator);
387 TotalVolume += boneSDF.BBoxVolume;
388 averageScale += boneSDF.Scale;
389 averagePosition += boneSDF.Position;
390 boneSDF.ManipulatorID = ID;
391 SDFObjectList.Add(boneSDF);
392 }
393
394 sdf.Position = averagePosition / skinnedMeshSDF.BoneSDFList.Count;
395 sdf.Scale = averageScale / skinnedMeshSDF.BoneSDFList.Count;
396 sdf.TotalGroupVolume = TotalVolume;
397 }
398
399 if (manipulator is ZibraLiquidEmitter)
400 manipulator.CurrentInteractionMode = Manipulator.InteractionMode.OnlySelectedParticleSpecies;
401
402 switch (manipulator.CurrentInteractionMode)
403 {
404 case Manipulator.InteractionMode.AllParticleSpecies:
405 manip.ParticleSpecies = 0;
406 break;
407 case Manipulator.InteractionMode.OnlySelectedParticleSpecies:
408 manip.ParticleSpecies = 1 + manipulator.ParticleSpecies;
409 break;
410 case Manipulator.InteractionMode.ExceptSelectedParticleSpecies:
411 manip.ParticleSpecies = -(1 + manipulator.ParticleSpecies);
412 break;
413 }
414
415 if (manipulator is ZibraLiquidEmitter)
416 {
417 ZibraLiquidEmitter emitter = manipulator as ZibraLiquidEmitter;
418
419 float particlesPerSec = emitter.VolumePerSimTime / parent.NodeSize / parent.NodeSize /
420 parent.NodeSize * parent.SolverParameters.ParticleDensity;
421
422 manip.AdditionalData0.x = Mathf.Floor(particlesPerSec * deltaTime);
423 manip.ParticleSpecies = manipulator.ParticleSpecies;
424 }
425
426 if (manipulator is ZibraLiquidVoid)
427 {
428 ZibraLiquidVoid liquidVoid = manipulator as ZibraLiquidVoid;
429
430 // Deletion percentage per frame is set in % / per "ZibraLiquid.DEFAULT_SIMULATION_TIME_SCALE" units
431 // of simulation time. 1 real second = (parent.SimulationTimeScale) units of simulation time. For
432 // example, deletePercentagePerDefaultSimulationTime = 0.5, parent.SimulationTimeScale = 100, then
433 // after 1 real second there should be only (50%) ^ (100 / 40) = 17% particles left. Therefore, to
434 // calculate percentage per frame:
435 // percentageThatWillLeftPerFrame = pow(deletePercentage, FPS * (1 /
436 // simulationTimeScaleDifference)),
437 // and
438 // deletionPercentagePerFrame = 1 - pow(1 - deletePercentage, 1/FPS *
439 // simulationTimeScaleDifference)
440 float simulationTimeScaleDifference =
441 ZibraLiquid.DEFAULT_SIMULATION_TIME_SCALE / parent.SimulationTimeScale;
442
443 float deltaTimeInRealSeconds = deltaTime / parent.SimulationTimeScale;
444 float deletionPercentagePerFrame =
445 1.0f - Mathf.Pow(1.0f - liquidVoid.DeletePercentage,
446 deltaTimeInRealSeconds * simulationTimeScaleDifference);
447
448 manip.AdditionalData0.x = deletionPercentagePerFrame;
449 }
450
451 manip.SDFObjectID = SDFObjectList.Count;
452 sdf.ManipulatorID = ID;
453 SDFObjectList.Add(sdf);
454 ManipulatorParams.Add(manip);
455 ID++;
456 }
457
458 Elements = Manipulators.Count;
459 }

461 private static float INT2Float(int a)
462 {
463 const float MAX_INT = 2147483647.0f;
464 const float F2I_MAX_VALUE = 5000.0f;
465 const float F2I_SCALE = (MAX_INT / F2I_MAX_VALUE);
466
467 return a / F2I_SCALE;
468 }

475 private Vector3 Simulation2Local(Vector3 pos, ZibraLiquid parent)
476 {
477 return new Vector3(parent.ContainerSize.x * pos.x / parent.GridSize.x - parent.ContainerSize.x * 0.5f,
478 parent.ContainerSize.y * pos.y / parent.GridSize.y - parent.ContainerSize.y * 0.5f,
479 parent.ContainerSize.z * pos.z / parent.GridSize.z - parent.ContainerSize.z * 0.5f);
480 }
481
482 private Vector3 Simulation2World(Vector3 pos, ZibraLiquid parent)
483 {
484 return Simulation2Local(pos, parent) + parent.transform.position;
485 }

487 private Vector3 EncodedSimulationSpaceToWorldSpace(Vector3 encodedPos, ZibraLiquid parent,
488 bool inverted = false)
489 {
490 Vector3 simulationPosition = Vector3.Scale(encodedPos / ((float)Int32.MaxValue), parent.GridSize);
491
492 if (inverted)
493 {
494 simulationPosition = parent.GridSize - simulationPosition;
495 }
496
497 return Simulation2World(simulationPosition, parent);
498 }

503 public void UpdateStatistics(ZibraLiquid parent, Int32[] data, List<Manipulator> curManipulators,
504 DataStructures.ZibraLiquidSolverParameters solverParameters,
505 List<ZibraLiquidCollider> sdfObjects)
506 {
507 int id = 0;
508 foreach (var manipulator in Manipulators)
509 {
510 if (manipulator == null)
511 continue;
512
513 Vector3 Force = Mathf.Exp(4.0f * solverParameters.ForceInteractionStrength) *
514 new Vector3(INT2Float(data[GetStatIndex(id, 0)]), INT2Float(data[GetStatIndex(id, 1)]),
515 INT2Float(data[GetStatIndex(id, 2)]));
516 Vector3 Torque = Mathf.Exp(4.0f * solverParameters.ForceInteractionStrength) *
517 new Vector3(INT2Float(data[GetStatIndex(id, 3)]), INT2Float(data[GetStatIndex(id, 4)]),
518 INT2Float(data[GetStatIndex(id, 5)]));
519
520 switch (manipulator.GetManipulatorType())
521 {
522 default:
523 break;
524 case Manipulator.ManipulatorType.Emitter:
525 ZibraLiquidEmitter emitter = manipulator as ZibraLiquidEmitter;
526 emitter.CreatedParticlesPerFrame = data[GetStatIndex(id, 0)];
527 emitter.CreatedParticlesTotal += emitter.CreatedParticlesPerFrame;
528 break;
529 case Manipulator.ManipulatorType.Void:
530 ZibraLiquidVoid zibravoid = manipulator as ZibraLiquidVoid;
531 zibravoid.DeletedParticleCountPerFrame = data[GetStatIndex(id, 0)];
532 zibravoid.DeletedParticleCountTotal += zibravoid.DeletedParticleCountPerFrame;
533 break;
534 case Manipulator.ManipulatorType.Detector:
535 ZibraLiquidDetector zibradetector = manipulator as ZibraLiquidDetector;
536 zibradetector.ParticlesInside = data[GetStatIndex(id, 0)];
537
538 if (zibradetector.ParticlesInside > 0)
539 {
540 // Decode bounding box position and convert them to world space.
541 zibradetector.BoundingBoxMin = EncodedSimulationSpaceToWorldSpace(
542 new Vector3(data[GetStatIndex(id, 1)], data[GetStatIndex(id, 2)],
543 data[GetStatIndex(id, 3)]),
544 parent,
545 true // BoundingBoxMin is inverted to simulate atomic min in the compute shader.
546 );
547 zibradetector.BoundingBoxMax = EncodedSimulationSpaceToWorldSpace(
548 new Vector3(data[GetStatIndex(id, 4)], data[GetStatIndex(id, 5)],
549 data[GetStatIndex(id, 6)]),
550 parent);
551 }
552 else
553 {
554 zibradetector.BoundingBoxMin = new Vector3(0.0f, 0.0f, 0.0f);
555 zibradetector.BoundingBoxMax = new Vector3(0.0f, 0.0f, 0.0f);
556 }
557
558 break;
559 case Manipulator.ManipulatorType.NeuralCollider:
560 case Manipulator.ManipulatorType.AnalyticCollider:
561 ZibraLiquidCollider collider = manipulator as ZibraLiquidCollider;
562 collider.ApplyForceTorque(Force, Torque);
563 break;
564 }
565#if UNITY_EDITOR
566 manipulator.NotifyChange();
567#endif
568
569 id++;
570 }
571 }

577 public void UpdateConst(List<Manipulator> curManipulators, List<ZibraLiquidCollider> colliders)
578 {
579 HeightmapBlit = GetComponent<ZibraLiquidMaterialParameters>().HeightmapBlit;
580
581 Manipulators = new List<Manipulator>();
582
583 NeuralSDFs = new Dictionary<ZibraHash128, NeuralSDF>();
584 TextureHashMap = new Dictionary<ZibraHash128, int>();
585
586 // add all colliders to the manipulator list
587 foreach (var manipulator in curManipulators)
588 {
589 if (manipulator == null)
590 continue;
591
592 var sdf = manipulator.GetComponent<SDFObject>();
593 if (sdf == null)
594 {
595 Debug.LogWarning("Manipulator " + manipulator.gameObject.name + " missing sdf and is disabled.");
596 continue;
597 }
598
599 NeuralSDF neuralSDF = sdf as NeuralSDF;
600 if (neuralSDF != null && !neuralSDF.ObjectRepresentation.HasRepresentationV3)
601 {
602 Debug.LogWarning("NeuralSDF in " + manipulator.gameObject.name +
603 " was not generated. Manipulator is disabled.");
604 continue;
605 }
606
607 TerrainSDF terrainSDF = sdf as TerrainSDF;
608 if (terrainSDF != null)
609 {
610 Debug.LogWarning("TerrainSDF was used in " + manipulator.gameObject.name +
611 " which is not supported. TerrainSDF only supported in colliders. " +
612 "Manipulator is disabled.");
613 continue;
614 }
615
616 if (sdf is SkinnedMeshSDF)
617 {
618 SkinnedMeshSDF skinnedMeshSDF = sdf as SkinnedMeshSDF;
619 if (!skinnedMeshSDF.HasRepresentation())
620 {
621 Debug.LogWarning("SkinnedMeshSDF in " + manipulator.gameObject.name +
622 " was not generated. Manipulator is disabled.");
623 continue;
624 }
625 }
626
627 if (sdf is SkinnedMeshSDF)
628 {
629 SkinnedMeshSDF skinnedMeshSDF = sdf as SkinnedMeshSDF;
630 if (!skinnedMeshSDF.HasRepresentation())
631 {
632 Debug.LogWarning("SkinnedMeshSDF in " + manipulator.gameObject.name +
633 " was not generated. Manipulator is disabled.");
634 continue;
635 }
636 }
637
638 Manipulators.Add(manipulator);
639 }
640
641 // add all colliders to the manipulator list
642 foreach (var manipulator in colliders)
643 {
644 if (manipulator == null)
645 continue;
646
647 var sdf = manipulator.GetComponent<SDFObject>();
648 if (sdf == null)
649 {
650 Debug.LogWarning("Collider " + manipulator.gameObject.name + " missing sdf and is disabled.");
651 continue;
652 }
653
654 if (sdf is NeuralSDF)
655 {
656 NeuralSDF neuralSDF = sdf as NeuralSDF;
657 if (!neuralSDF.ObjectRepresentation.HasRepresentationV3)
658 {
659 Debug.LogWarning("NeuralSDF in " + manipulator.gameObject.name +
660 " was not generated. Collider is disabled.");
661 continue;
662 }
663 }
664
665 if (sdf is SkinnedMeshSDF)
666 {
667 SkinnedMeshSDF skinnedMeshSDF = sdf as SkinnedMeshSDF;
668 if (!skinnedMeshSDF.HasRepresentation())
669 {
670 Debug.LogWarning("SkinnedMeshSDF in " + manipulator.gameObject.name +
671 " was not generated. Collider is disabled.");
672 continue;
673 }
674 }
675
676 TerrainSDF terrainSDF = sdf as TerrainSDF;
677 if (terrainSDF != null)
678 {
679 Terrain terrain = terrainSDF.GetComponent<Terrain>();
680 if (terrain == null)
681 {
682 Debug.LogWarning("TerrainSDF in " + manipulator.gameObject.name +
683 " is missing Terrain component. Collider is disabled.");
684 continue;
685 }
686 if (terrain.terrainData == null)
687 {
688 Debug.LogWarning("TerrainSDF in " + manipulator.gameObject.name +
689 " is missing TerrainData in Terrain component. Collider is disabled.");
690 continue;
691 }
692 }
693
694 Manipulators.Add(manipulator);
695 }
